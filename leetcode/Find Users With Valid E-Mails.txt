select u.*
from users u
where instr(mail,'@leetcode.com')<>0 and
     replace(translate(lower(mail),'abcdefghijklmnopqrstuvwxyz1234567890_-@.','                                        '),' ','') is null and
     instr(replace(mail,'@leetcode.com',''),'@')=0 and
     substr(mail,1,1) not in ('1','2','3','4','5','6','7','8','9','0','_','-','.')
